00030 DEFUN_DLD(
00031     arsel, args, nargout,
00032     "\t[A, mu, sigma2eps, eff_sigma2x, T0] = arsel (d, submean=true, order=512)\n"
00033     "\tAutomatically fit an autoregressive model to an input signal.\n"
00034     "\t\n"
00035     "\tUse ar::burg_method followed by ar::best_model<CIC<Burg<MeanHandling > >\n"
00036     "\tto obtain the most likely autoregressive process for input signal d.\n"
00037     "\tThe sample mean of d will be subtracted whenever submean is true.\n"
00038     "\tModel orders zero through min(size(d), order) will be considered.\n"
00039     "\t\n"
00040     "\tThe filter()-ready process parameters are returned in A, the sample mean\n"
00041     "\tin mu, and the innovation variance \\sigma^2_\\epsilon in sigma2eps.\n"
00042     "\tGiven the observed autocorrelation structure, a decorrelation time T0 is\n"
00043     "\tcomputed and used to estimate the effective signal variance eff_sigma2x.\n"
00044     "\t\n"
00045     "\tOne may simulate N samples from the fitted process by afterwards calling:\n"
00046     "\t\n"
00047     "\t\tx = mu + filter([1], A, sqrt(sigma2eps)*randn(N,1));\n"
00048     "\t\n"
00049 )
00050 {
00051     std::size_t order = 512;
00052     bool        submean = true;
00053     RowVector   data;
00054     switch (args.length())  // Cases deliberately fall through
00055     {
00056         case 3:
00057             order = args(2).ulong_value();
00058         case 2:
00059             submean = args(1).bool_value();
00060         case 1:
00061             data = args(0).row_vector_value();
00062             break;
00063         default:
00064             error("Invalid call to arsel.  Correct usage is: ");
00065         case 0:
00066             print_usage();
00067             return octave_value();
00068     }
00069 
00070     // How much data will we be processing?
00071     const size_t N = data.dims().numel();
00072     if (!N) {
00073         error("arsel: input signal d must have nonzero length");
00074         return octave_value();
00075     }
00076 
00077     // Use burg_method to estimate a hierarchy of AR models from input data
00078     double mu;
00079     std::vector<double> params, sigma2e, gain, autocor;
00080     params .reserve(order*(order + 1)/2);
00081     sigma2e.reserve(order + 1);
00082     gain   .reserve(order + 1);
00083     autocor.reserve(order + 1);
00084     ar::burg_method(data.fortran_vec(), data.fortran_vec() + N,
00085                     mu, order,
00086                     std::back_inserter(params),
00087                     std::back_inserter(sigma2e),
00088                     std::back_inserter(gain),
00089                     std::back_inserter(autocor),
00090                     submean, /* output hierarchy? */ true);
00091 
00092 
00093     // Keep only best model according to CIC accounting for subtract_mean.
00094     if (submean) {
00095         ar::best_model<ar::CIC<ar::Burg<ar::mean_subtracted> > >(
00096                 N, params, sigma2e, gain, autocor);
00097     } else {
00098         ar::best_model<ar::CIC<ar::Burg<ar::mean_retained> > >(
00099                 N, params, sigma2e, gain, autocor);
00100     }
00101 
00102     // Compute decorrelation time from the estimated autocorrelation model
00103     const double T0  = ar::decorrelation_time(N, ar::autocorrelation(
00104                 params.begin(), params.end(), gain[0], autocor.begin()),
00105                 /* abs(rho) */ true);
00106 
00107     // Prepare autoregressive process parameters for Octave's filter function
00108     RowVector A(params.size() + 1, /* leading one */ 1);
00109     std::copy(params.begin(), params.end(), A.fortran_vec() + 1);
00110 
00111     // Prepare output: [A, mu, sigma2eps, eff_sigma2x, T0]
00112     octave_value_list retval;
00113     switch (nargout)  // Cases deliberately fall through
00114     {
00115         default:
00116             warning("arsel: too many output values requested");
00117         case 5:
00118             retval.prepend(T0);
00119         case 4:
00120             retval.prepend((N*gain[0]*sigma2e[0]) / (N - T0));
00121         case 3:
00122             retval.prepend(sigma2e[0]);
00123         case 2:
00124             retval.prepend(mu);
00125         case 1:
00126             retval.prepend(A);
00127             break;
00128         case 0:
00129             warning("arsel: no output values requested");
00130     }
00131 
00132     // Return results iff no errors were detected
00133     return error_state ? octave_value_list() : retval;
