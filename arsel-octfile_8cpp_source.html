41 DEFUN_DLD(
42  arsel, args, nargout,
43  "\t[A, mu, sigma2eps, eff_var, eff_N, T0] = arsel (d, submean, absrho, maxorder)\n"
44  "\tAutomatically fit an autoregressive model to an input signal.\n"
45  "\t\n"
46  "\tUse ar::burg_method and ar::best_model<CIC<Burg<MeanHandling > > to\n"
47  "\tobtain the most likely autoregressive process for input signal d.\n"
48  "\tThe sample mean of d will be subtracted whenever submean is true.\n"
49  "\tThe absolute value of the autocorrelation function will be used\n"
50  "\tin computing the decorrelation time T0 whenever absrho is true.\n"
51  "\tModel orders zero through min(size(d), maxorder) will be considered.\n"
52  "\t\n"
53  "\tThe filter()-ready process parameters are returned in A, the sample mean\n"
54  "\tin mu, and the innovation variance \\sigma^2_\\epsilon in sigma2eps.\n"
55  "\tGiven the observed autocorrelation structure, a decorrelation time T0 is\n"
56  "\tcomputed and used to estimate the effective signal variance eff_var.\n"
57  "\tThe number of effectively independent samples is returned in eff_N.\n"
58  "\t\n"
59  "\tOne may simulate N samples from the fitted process by calling:\n"
60  "\t\n"
61  "\t\tx = mu + filter([1], A, sqrt(sigma2eps)*randn(N,1));\n"
62  "\t\n"
63  "\tWhen not supplied, submean defaults to " STRINGIFY(DEFAULT_SUBMEAN) ".\n"
64  "\tWhen not supplied, absrho defaults to " STRINGIFY(DEFAULT_ABSRHO) ".\n"
65  "\tWhen not supplied, maxorder defaults to " STRINGIFY(DEFAULT_MAXORDER) ".\n"
66  "\tWhen no outputs are requested, a struct with all outputs is returned.\n"
67 )
68 {
69  std::size_t maxorder = DEFAULT_MAXORDER;
70  bool absrho = DEFAULT_ABSRHO;
71  bool submean = DEFAULT_SUBMEAN;
72  RowVector data;
73  switch (args.length())
74  {
75  case 4:
76  maxorder = args(3).ulong_value();
77  case 3:
78  absrho = args(2).bool_value();
79  case 2:
80  submean = args(1).bool_value();
81  case 1:
82  data = args(0).row_vector_value();
83  break;
84  default:
85  error("Invalid call to arsel. Correct usage is: ");
86  case 0:
87  print_usage();
88  return octave_value();
89  }
90 
91  // How much data will we be processing?
92  const size_t N = data.dims().numel();
93  if (!N)
94  {
95  error("arsel: input signal d must have nonzero length");
96  return octave_value();
97  }
98 
99  // Use burg_method to estimate a hierarchy of AR models from input data
100  double mu;
101  std::vector<double> params, sigma2e, gain, autocor;
102  params .reserve(maxorder*(maxorder + 1)/2);
103  sigma2e.reserve(maxorder + 1);
104  gain .reserve(maxorder + 1);
105  autocor.reserve(maxorder + 1);
106  ar::burg_method(data.fortran_vec(), data.fortran_vec() + N,
107  mu, maxorder,
108  std::back_inserter(params),
109  std::back_inserter(sigma2e),
110  std::back_inserter(gain),
111  std::back_inserter(autocor),
112  submean, /* output hierarchy? */ true);
113 
114  // Keep only best model according to CIC accounting for subtract_mean.
115  if (submean)
116  {
117  ar::best_model<ar::CIC<ar::Burg<ar::mean_subtracted> > >(
118  N, params, sigma2e, gain, autocor);
119  }
120  else
121  {
122  ar::best_model<ar::CIC<ar::Burg<ar::mean_retained> > >(
123  N, params, sigma2e, gain, autocor);
124  }
125 
126  // Compute decorrelation time from the estimated autocorrelation model
127  double T0 = std::numeric_limits<double>::quiet_NaN();
128  if (nargout == 0 || nargout > 2)
129  {
130  ar::predictor<double> p = ar::autocorrelation(
131  params.begin(), params.end(), gain[0], autocor.begin());
132  T0 = ar::decorrelation_time(N, p, absrho);
133  }
134 
135  // Prepare autoregressive process parameters for Octave's filter function
136  RowVector A(params.size() + 1, /* leading one */ 1);
137  std::copy(params.begin(), params.end(), A.fortran_vec() + 1);
138 
139  // Prepare output: [A, mu, sigma2eps, eff_var, eff_N, T0]
140  // Unbiased effective variance expression from [Trenberth1984]
141  octave_value_list result;
142  std::list<std::string> names;
143  switch (nargout == 0 ? std::numeric_limits<int>::max() : nargout)
144  {
145  default:
146  if (nargout) warning("arsel: too many output values requested");
147  case 6:
148  names.push_front("T0");
149  result.prepend(T0);
150  case 5:
151  names.push_front("eff_N");
152  result.prepend(N / T0);
153  case 4:
154  names.push_front("eff_var");
155  result.prepend((N*gain[0]*sigma2e[0]) / (N - T0));
156  case 3:
157  names.push_front("sigma2eps");
158  result.prepend(sigma2e[0]);
159  case 2:
160  names.push_front("mu");
161  result.prepend(mu);
162  case 1:
163  names.push_front("A");
164  result.prepend(A);
165  break;
166  case 0:
167  panic_impossible();
168  }
169 
170  // Either return requested results in assignment-friendly fashion
171  // or, when nothing was requested, return a struct to likely display.
172  if (nargout)
173  {
174  result.stash_name_tags(names);
175  }
176  else // nargout == 0
177  {
178  Octave_map m;
179  int i = 0;
180  while (names.size()) {
181  m.assign(names.front(), result(i++));
182  names.pop_front();
183  }
184  result.resize(1);
185  result(0) = m;
186  }
187 
188  // Provide no results whenever an error was detected
189  if (error_state)
190  {
191  warning("arsel: error detected; no results returned");
192  result.resize(0);
193  }
194 
195  return result;
196 }
