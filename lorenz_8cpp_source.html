102 static inline void lorenz(
103  const real beta, const real rho, const real sigma,
104  const real x, const real y, const real z,
105  real &dxdt, real &dydt, real &dzdt)
106 {
107  dxdt = sigma*(y - x);
108  dydt = x*(rho - z) - y;
109  dzdt = x*y - beta*z;
110 }

113 static void euler(
114  const real dt, const real beta, const real rho, const real sigma,
115  long double &t, real &x, real &y, real &z)
116 {
117  real dxdt, dydt, dzdt;
118  lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
119  x += dt*dxdt;
120  y += dt*dydt;
121  z += dt*dzdt;
122  t += dt;

129 static void tvd_rk2(
130  const real dt, const real beta, const real rho, const real sigma,
131  long double &t, real &x, real &y, real &z)
132 {
133  real dxdt, dydt, dzdt;
134 
135  lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
136  real u1x = x + dt*dxdt;
137  real u1y = y + dt*dydt;
138  real u1z = z + dt*dzdt;
139 
140  lorenz(beta, rho, sigma, u1x, u1y, u1z, dxdt, dydt, dzdt);
141  x = (x + u1x + dt*dxdt)/2;
142  y = (y + u1y + dt*dydt)/2;
143  z = (z + u1z + dt*dzdt)/2;
144  t += dt;
145 }

151 static void tvd_rk3(
152  const real dt, const real beta, const real rho, const real sigma,
153  long double &t, real &x, real &y, real &z)
154 {
155  real dxdt, dydt, dzdt;
156 
157  lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
158  real u1x = x + dt*dxdt;
159  real u1y = y + dt*dydt;
160  real u1z = z + dt*dzdt;
161 
162  lorenz(beta, rho, sigma, u1x, u1y, u1z, dxdt, dydt, dzdt);
163  real u2x = (3*x + u1x + dt*dxdt)/4;
164  real u2y = (3*y + u1y + dt*dydt)/4;
165  real u2z = (3*z + u1z + dt*dzdt)/4;
166 
167  lorenz(beta, rho, sigma, u2x, u2y, u2z, dxdt, dydt, dzdt);
168  x = (x + 2*u2x + 2*dt*dxdt)/3;
169  y = (y + 2*u2y + 2*dt*dydt)/3;
170  z = (z + 2*u2z + 2*dt*dzdt)/3;
171  t += dt;
172 }

177 static void std_rk4(
178  const real dt, const real beta, const real rho, const real sigma,
179  long double &t, real &x, real &y, real &z)

181  real dxdt, dydt, dzdt;

183  lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
184  real ux = x + dt*dxdt/2;
185  real uy = y + dt*dydt/2;

193  ux = x + dt*dxdt/2;

199  dz += dt*dzdt/3;

203  uy = y + dt*dydt;

208  dz += dt*dzdt/3;

212  y += (dy + dt*dydt/6);
213  z += (dz + dt*dzdt/6);

233  real burn = 500;
234  real dt = 0.01;

236  real rho = 28;

288  if (opts[BURN ]) burn = strtod(opts[BURN ].last()->arg, NULL);
289  if (opts[DT ]) dt = strtod(opts[DT ].last()->arg, NULL);
290  if (opts[EVERY ]) every = strtol(opts[EVERY ].last()->arg, NULL, 10);
291  if (opts[RHO ]) rho = strtod(opts[RHO ].last()->arg, NULL);

300  }
301  }
302 
303  // Discard 0 <= t < burn
304  switch (scheme) { // Switch designed to avoid spurious jumps
305  default: cerr << "Sanity failure: unknown scheme at "
306  << __FILE__ << ":" << __LINE__ << '\n';
307  return EXIT_FAILURE;
308  case RK1: while (t < burn) euler (dt, beta, rho, sigma, t, x, y, z);
309  break;
310  case RK2: while (t < burn) tvd_rk2(dt, beta, rho, sigma, t, x, y, z);

312  case RK3: while (t < burn) tvd_rk3(dt, beta, rho, sigma, t, x, y, z);

321  do {
322 
323  cout << t << '\t' << x << '\t' << y << '\t' << z;
324  if (more) {

326  << '\t' << y*y << '\t' << y*z
327  << '\t' << z*z;
328  }
329  cout << '\n';
330 
331  switch (scheme) {
332  default:
333  cerr << "Sanity failure: unknown scheme at "
334  << __FILE__ << ":" << __LINE__ << '\n';
335  return EXIT_FAILURE;
336  case RK1:
337  for (long i = 0; i < every; ++i) {
338  euler (dt, beta, rho, sigma, t, x, y, z);
339  }
340  break;
341  case RK2:
342  for (long i = 0; i < every; ++i) {
343  tvd_rk2(dt, beta, rho, sigma, t, x, y, z);
344  }
345  break;
346  case RK3:
347  for (long i = 0; i < every; ++i) {
348  tvd_rk3(dt, beta, rho, sigma, t, x, y, z);
349  }
350  break;
351  case RK4:
352  for (long i = 0; i < every; ++i) {
353  std_rk4(dt, beta, rho, sigma, t, x, y, z);
354  }
355  break;
356  }
357 
358  } while (t < tfinal);
359 
360  return EXIT_SUCCESS;
361 }
