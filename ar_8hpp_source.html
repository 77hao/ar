00501     return N;

01046         // \eta_i   = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
01047         const value neg_eta   = inner_product(
01048                 ehat.begin(), ehat.end(), a_first, value(a_first[i]));
01049 
01050         // \gamma_i = -\rho_{i+1}    - \tilde{g}_i \hat{r}_i
01051         const value neg_gamma = inner_product(
01052                 g.begin(), g.end(), rhat_first, value(r_first[i]));
01053 
01054         /*
01055          * s_{i+1} = \bigl(\begin{smallmatrix}
01056          *              s_i + (\theta_i/\lambda_i) \hat{e}_i \\
01057          *              \theta_i/\lambda_i
01058          *          \end{smallmatrix}\bigr)
01059          *
01060          * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
01061          *                     \eta_i/\lambda_i \\
01062          *                     \hat{e}_i + (\eta_i/\lambda_i) g_i
01063          *                 \end{smallmatrix}\bigr)
01064          *
01065          * g_{i+1} = \bigl(\begin{smallmatrix}
01066          *               g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
01067          *               \gamma_i/\lambda_i
01068          *           \end{smallmatrix}\bigr)
01069          */
01070         const value theta_by_lambda = -neg_theta/lambda;
01071         const value   eta_by_lambda = -neg_eta  /lambda;
01072         const value gamma_by_lambda = -neg_gamma/lambda;
01073         next_ehat.clear();
01074         next_ehat.push_back(eta_by_lambda);
01075         for (size j = 0; j < i; ++j)
01076         {
01077             s[j] += theta_by_lambda*ehat[j];
01078             next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
01079             g[j] += gamma_by_lambda*ehat[j];
01080         }
01081         s.push_back(theta_by_lambda);
01082         g.push_back(gamma_by_lambda);
01083         ehat.swap(next_ehat);
01084 
01085         // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
01086         lambda -= neg_eta*neg_gamma/lambda;
01087     }
01088 
01089     // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
01090     // Computing g_n above was unnecessary but the incremental expense is small
01091     {
01092         reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
01093 
01094         // \theta_n =  \delta_{n+1}  - \tilde{s}_n \hat{r}_n
01095         const value neg_theta = inner_product(
01096                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
01097 
01098         /*
01099          * s_{n+1} = \bigl(\begin{smallmatrix}
01100          *              s_n + (\theta_n/\lambda_n) \hat{e}_n \\
01101          *              \theta_n/\lambda_n
01102          *          \end{smallmatrix}\bigr)
01103          */
01104         const value theta_by_lambda = -neg_theta/lambda;
01105         for (size j = 0; j < n; ++j)
01106         {
01107             s[j] += theta_by_lambda*ehat[j];
01108         }
01109         s.push_back(theta_by_lambda);
01110     }
01111 
01112     // Output solution
01113     copy(s.begin(), s.end(), s_first);
01114 }
01115

02007     typedef typename iterator_traits<iterator>::value_type      value;

02009     // Short circuit on trivial input
02010     if (first == last)
02011         return -1;
02012 
02013     // Handle first iteration without comparison as AICC blows up on N == 1
02014     value best_val = evaluate<Criterion>(*first++, N, ordfirst);
02015     difference best_pos = 0, dist = 0;
02016 
02017     // Scan through rest of candidates (up to order N-1) updating best as we go
02018     while (first != last && static_cast<Integer1>(++dist) < N)
02019     {
02020         value candidate = evaluate<Criterion>(*first++, N, dist + ordfirst);
02021         *crit++ = candidate;
02022 
02023         if (candidate < best_val) {

02025             best_pos = dist;
02026         }
02027     }
02028

02067     using std::advance;
02068     using std::copy_backward;
02069 
02070     // Ensure all inputs have conformant sizes
02071     assert(sigma2e.size() > 0);
02072     assert(params .size() == (sigma2e.size()-1)*(sigma2e.size())/2);
02073     assert(gain   .size() == sigma2e.size());
02074     assert(autocor.size() == sigma2e.size());
02075 
02076     // Find the best model index according to the given criterion
02077     const typename Sequence1::difference_type best = evaluate_models<Criterion>(
02078             N, 0u, sigma2e.begin(), sigma2e.end(), crit);
02079 
02080     // Now trim away everything and leaving only the best model in Sequences...
02081 
02082     // ...first in params...
02083     {
02084         // Best parameters might overlap beginning of params so copy_backwards.
02085         // AR(0) is trivial and AR(1) starts at params.begin(), hence off by 1.
02086         typename Sequence1::iterator first  = params.begin();
02087         typename Sequence1::iterator last   = params.begin();
02088         typename Sequence1::iterator result = params.begin();
02089         advance(first,  ((best-1)*best)/2       );
02090         advance(last,   ((best-1)*best)/2 + best);
02091         advance(result,                     best);
02092         copy_backward(first, last, result);
02093         params.resize(best);
02094     }
02095 
02096     // ...next in sigma2e...
02097     {
02098         typename Sequence2::iterator cursor = sigma2e.begin();
02099         advance(cursor, best);
02100         *sigma2e.begin() = *cursor;
02101         sigma2e.resize(1);
02102     }
02103 
02104     // ...next in gain...
02105     {
02106         typename Sequence2::iterator cursor = gain.begin();
02107         advance(cursor, best);
02108         *gain.begin() = *cursor;
02109         gain.resize(1);
02110     }
02111 
02112     // ...and last in autocor...
02113     autocor.resize(best + 1);
02114 
02115     // ...but notice that resizing does not free capacity for std::vectors.
02116 
02117     return best;
02118 }
