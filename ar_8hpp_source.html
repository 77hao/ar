00621     using std::copy;
00622     using std::distance;
00623     using std::fill;
00624     using std::inner_product;
00625     using std::min;
00626     using std::minus;
00627     using std::size_t;
00628 
00629     // Initialize f from [data_first, data_last) and fix number of samples
00630     f.assign(data_first, data_last);
00631     const size_t N = f.size();
00632 
00633     // Stably compute the incoming data's mean and population variance
00634     mean = 0;
00635     Value sigma2e = 0;
00636     welford_variance_population(f.begin(), f.end(), mean, sigma2e);
00637 
00638     // When requested, subtract the just-computed mean from the data.
00639     // Adjust, if necessary, to make sigma2e the second moment.
00640     if (subtract_mean)
00641     {
00642         transform(f.begin(), f.end(), f.begin(), bind2nd(minus<Value>(), mean));
00643     }
00644     else
00645     {
00646         sigma2e += mean*mean;
00647     }
00648 
00649     // At most maxorder N-1 can be fit from N samples.  Beware N is unsigned.
00650     maxorder = (N == 0) ? 0 : min(static_cast<size_t>(maxorder), N-1);
00651 
00652     // Output sigma2e and gain for a zeroth order model, if requested.
00653     Value gain = 1;
00654     if (hierarchy || maxorder == 0)
00655     {
00656         *sigma2e_first++ = sigma2e;
00657         *gain_first++    = gain;
00658     }
00659 
00660     // Initialize and perform Burg recursion
00661     if (maxorder) b = f;  // Copy iff non-trivial work required
00662     Ak.assign(maxorder + 1, Value(0));
00663     Ak[0] = 1;
00664     ac.clear();
00665     ac.reserve(maxorder);
00666     for (size_t kp1 = 1; kp1 <= maxorder; ++kp1)
00667     {
00668         // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
00669         // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
00670         // By the recurrence, Ak[kp1] will also be the reflection coefficient
00671         Value mu = -2 * negative_half_reflection_coefficient<Value>(
00672                 f.begin() + kp1, f.end(), b.begin());
00673 
00674         sigma2e *= (1 - mu*mu);
00675         for (size_t n = 0; n <= kp1/2; ++n)
00676         {
00677             Value t1 = Ak[n] + mu*Ak[kp1 - n];
00678             Value t2 = Ak[kp1 - n] + mu*Ak[n];
00679             Ak[n] = t1;
00680             Ak[kp1 - n] = t2;
00681         }
00682 
00683         // Update the gain per Broersen 2006 equation (5.25)
00684         gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
00685 
00686         // Compute and output the next autocorrelation coefficient
00687         // See Broersen 2006 equations (5.28) and (5.31) for details
00688         ac.push_back(-inner_product(ac.rbegin(), ac.rend(),

00690 
00691         // Output parameters and the input and output variances when requested
00692         if (hierarchy || kp1 == maxorder)
00693         {

00695                                 params_first);
00696             *sigma2e_first++ = sigma2e;
00697             *gain_first++    = gain;
00698         }
00699 
00700         // Update f and b for the next iteration if another remains
00701         if (kp1 < maxorder)
00702         {
00703             for (size_t n = 0; n < N - kp1; ++n)
00704             {
00705                 Value t1 = f[n + kp1] + mu*b[n];
00706                 Value t2 = b[n] + mu*f[n + kp1];
00707                 f[n + kp1] = t1;
00708                 b[n] = t2;
00709             }
00710         }
00711     }
00712 
00713     // Output the lag [0,maxorder] autocorrelation coefficients in single pass
00714     *autocor_first++ = 1;
00715     copy(ac.begin(), ac.end(), autocor_first);
00716 
00717     // Return the number of values processed in [data_first, data_last)
00718     return N;
00719 }

01221 template<class RandomAccessIterator,
01222          class InputIterator,
01223          class OutputIterator>
01224 void zohar_linear_solve(RandomAccessIterator a_first,
01225                         RandomAccessIterator a_last,
01226                         RandomAccessIterator r_first,
01227                         InputIterator        d_first,
01228                         OutputIterator       s_first)
01229 {
01230     using std::copy;
01231     using std::distance;
01232     using std::inner_product;
01233     using std::invalid_argument;
01234     using std::iterator_traits;
01235     using std::reverse_iterator;
01236     using std::vector;
01237 
01238     // Tildes indicate transposes while hats indicate reversed vectors.
01239 
01240     // InputIterator::value_type determines the working precision
01241     typedef typename iterator_traits<InputIterator>::value_type value_type;
01242     typedef vector<value_type> vector_type;
01243     typedef typename vector_type::size_type size_type;
01244 
01245     // Determine problem size using [a_first,a_last) and ensure nontrivial
01246     typename iterator_traits<RandomAccessIterator>::difference_type dist
01247             = distance(a_first, a_last);
01248     if (dist < 1) throw invalid_argument("distance(a_first, a_last) < 1");
01249     const size_type n = static_cast<size_type>(dist);
01250 
01251     // Allocate working storage and set initial values for recursion:
01252     vector_type s;    s   .reserve(n+1); s   .push_back( *d_first);
01253     vector_type ehat; ehat.reserve(n  ); ehat.push_back(-a_first[0]);
01254     vector_type g;    g   .reserve(n  ); g   .push_back(-r_first[0]);
01255     value_type lambda  = 1 - a_first[0]*r_first[0];
01256 
01257     // Though recursive updates to s and g can be done in-place, updates to
01258     // ehat seemingly require one additional vector for storage:
01259     //
01260     // "This sequence of computations is economical of storage.  It is only
01261     // necessary to retain quantities computed at level m - 1 until the
01262     // computations at level m are complete." [Trench1967, page 1504]
01263     vector_type next_ehat; next_ehat.reserve(n);
01264 
01265     // Recursion for i = {1, 2, ..., n - 1}:
01266     for (size_type i = 1; i < n; ++i)
01267     {
01268 
01269         reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
01270 
01271         // \theta_i =  \delta_{i+1}  - \tilde{s}_i \hat{r}_i
01272         const value_type neg_theta = inner_product(
01273                 s.begin(), s.end(), rhat_first, value_type(-(*++d_first)));
01274 
01275         // \eta_i   = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
01276         const value_type neg_eta   = inner_product(
01277                 ehat.begin(), ehat.end(), a_first, value_type(a_first[i]));
01278 
01279         // \gamma_i = -\rho_{i+1}    - \tilde{g}_i \hat{r}_i
01280         const value_type neg_gamma = inner_product(
01281                 g.begin(), g.end(), rhat_first, value_type(r_first[i]));
01282 
01283         /*
01284          * s_{i+1} = \bigl(\begin{smallmatrix}
01285          *              s_i + (\theta_i/\lambda_i) \hat{e}_i \\
01286          *              \theta_i/\lambda_i
01287          *          \end{smallmatrix}\bigr)
01288          *
01289          * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
01290          *                     \eta_i/\lambda_i \\
01291          *                     \hat{e}_i + (\eta_i/\lambda_i) g_i
01292          *                 \end{smallmatrix}\bigr)
01293          *
01294          * g_{i+1} = \bigl(\begin{smallmatrix}
01295          *               g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
01296          *               \gamma_i/\lambda_i
01297          *           \end{smallmatrix}\bigr)
01298          */
01299         const value_type theta_by_lambda = -neg_theta/lambda;
01300         const value_type   eta_by_lambda = -neg_eta  /lambda;
01301         const value_type gamma_by_lambda = -neg_gamma/lambda;
01302         next_ehat.clear();
01303         next_ehat.push_back(eta_by_lambda);
01304         for (size_type j = 0; j < i; ++j)
01305         {
01306             s[j] += theta_by_lambda*ehat[j];
01307             next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
01308             g[j] += gamma_by_lambda*ehat[j];
01309         }
01310         s.push_back(theta_by_lambda);
01311         g.push_back(gamma_by_lambda);
01312         ehat.swap(next_ehat);
01313 
01314         // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
01315         lambda -= neg_eta*neg_gamma/lambda;
01316     }
01317 
01318     // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
01319     // Computing g_n above was unnecessary but the incremental expense is small
01320     {
01321         reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
01322 
01323         // \theta_n =  \delta_{n+1}  - \tilde{s}_n \hat{r}_n
01324         const value_type neg_theta = inner_product(
01325                 s.begin(), s.end(), rhat_first, value_type(-(*++d_first)));
01326 
01327         /*
01328          * s_{n+1} = \bigl(\begin{smallmatrix}
01329          *              s_n + (\theta_n/\lambda_n) \hat{e}_n \\
01330          *              \theta_n/\lambda_n
01331          *          \end{smallmatrix}\bigr)
01332          */
01333         const value_type theta_by_lambda = -neg_theta/lambda;
01334         for (size_type j = 0; j < n; ++j)
01335         {
01336             s[j] += theta_by_lambda*ehat[j];
01337         }
01338         s.push_back(theta_by_lambda);
01339     }
01340 
01341     // Output solution
01342     copy(s.begin(), s.end(), s_first);
01343 }
