00412     f.assign(data_first, data_last);
00413     const size_t N = f.size();

00415     // Compute the incoming data's mean and centered sum of squares
00416     mean = 0;
00417     Value sigma2e = 0;
00418     welford_nvariance(f.begin(), f.end(), mean, sigma2e);
00419

00433     Value Dk = - f[0]*f[0] - f[N - 1]*f[N - 1] + 2*sigma2e;

00437     maxorder = (N == 0) ? 0 : min(static_cast<size_t>(maxorder), N-1);

00439     // Output sigma2e and gain for a zeroth order model, if requested.

00444     }

00462         for (size_t n = 0; n <= kp1/2; ++n)

00464             Value t1 = Ak[n] - mu*Ak[kp1 - n];
00465             Value t2 = Ak[kp1 - n] - mu*Ak[n];

00467             Ak[kp1 - n] = t2;
00468         }
00469 
00470         // Update the gain per Broersen 2006 equation (5.25)
00471         gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);

00485         }

00491             {
00492                 Value t1 = f[n + kp1] - mu*b[n];
00493                 Value t2 = b[n] - mu*f[n + kp1];
00494                 f[n + kp1] = t1;
00495                 b[n] = t2;
00496             }

00498         }
00499     }
00500 
00501     // Output the lag [0,maxorder] autocorrelation coefficients in single pass
00502     *autocor_first++ = 1;
00503     copy(ac.begin(), ac.end(), autocor_first);
00504 
00505     // Return the number of values processed in [data_first, data_last)
00506     return N;
00507 }

01039     vector next_ehat; next_ehat.reserve(n);

01044 
01045         reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
01046 
01047         // \theta_i =  \delta_{i+1}  - \tilde{s}_i \hat{r}_i
01048         const value neg_theta = inner_product(
01049                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));

01051         // \eta_i   = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
01052         const value neg_eta   = inner_product(
01053                 ehat.begin(), ehat.end(), a_first, value(a_first[i]));
01054 
01055         // \gamma_i = -\rho_{i+1}    - \tilde{g}_i \hat{r}_i
01056         const value neg_gamma = inner_product(
01057                 g.begin(), g.end(), rhat_first, value(r_first[i]));
01058 
01059         /*
01060          * s_{i+1} = \bigl(\begin{smallmatrix}
01061          *              s_i + (\theta_i/\lambda_i) \hat{e}_i \\
01062          *              \theta_i/\lambda_i
01063          *          \end{smallmatrix}\bigr)
01064          *
01065          * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
01066          *                     \eta_i/\lambda_i \\
01067          *                     \hat{e}_i + (\eta_i/\lambda_i) g_i
01068          *                 \end{smallmatrix}\bigr)
01069          *
01070          * g_{i+1} = \bigl(\begin{smallmatrix}
01071          *               g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
01072          *               \gamma_i/\lambda_i
01073          *           \end{smallmatrix}\bigr)
01074          */
01075         const value theta_by_lambda = -neg_theta/lambda;
01076         const value   eta_by_lambda = -neg_eta  /lambda;
01077         const value gamma_by_lambda = -neg_gamma/lambda;
01078         next_ehat.clear();
01079         next_ehat.push_back(eta_by_lambda);
01080         for (size j = 0; j < i; ++j)
01081         {
01082             s[j] += theta_by_lambda*ehat[j];
01083             next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
01084             g[j] += gamma_by_lambda*ehat[j];
01085         }
01086         s.push_back(theta_by_lambda);
01087         g.push_back(gamma_by_lambda);
01088         ehat.swap(next_ehat);
01089 
01090         // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
01091         lambda -= neg_eta*neg_gamma/lambda;
01092     }
01093 
01094     // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
01095     // Computing g_n above was unnecessary but the incremental expense is small
01096     {
01097         reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
01098 
01099         // \theta_n =  \delta_{n+1}  - \tilde{s}_n \hat{r}_n
01100         const value neg_theta = inner_product(
01101                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
01102 
01103         /*
01104          * s_{n+1} = \bigl(\begin{smallmatrix}
01105          *              s_n + (\theta_n/\lambda_n) \hat{e}_n \\
01106          *              \theta_n/\lambda_n
01107          *          \end{smallmatrix}\bigr)
01108          */
01109         const value theta_by_lambda = -neg_theta/lambda;
01110         for (size j = 0; j < n; ++j)
01111         {
01112             s[j] += theta_by_lambda*ehat[j];
01113         }
01114         s.push_back(theta_by_lambda);
01115     }
01116 
01117     // Output solution
01118     copy(s.begin(), s.end(), s_first);
01119 }
01120

02090         // AR(0) is trivial and AR(1) starts at params.begin(), hence off by 1.
02091         typename Sequence1::iterator first  = params.begin();
02092         typename Sequence1::iterator last   = params.begin();
02093         typename Sequence1::iterator result = params.begin();
02094         advance(first,  ((best-1)*best)/2       );

02096         advance(result,                     best);

02104         advance(cursor, best);
