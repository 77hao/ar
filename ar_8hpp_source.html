387 std::size_t burg_method(InputIterator data_first,
388  InputIterator data_last,
389  Value& mean,
390  std::size_t& maxorder,
391  OutputIterator1 params_first,
392  OutputIterator2 sigma2e_first,
393  OutputIterator3 gain_first,
394  OutputIterator4 autocor_first,
395  const bool subtract_mean,
396  const bool hierarchy,
397  Vector& f,
398  Vector& b,
399  Vector& Ak,
400  Vector& ac)
401 {
402  using std::bind2nd;
403  using std::copy;
404  using std::distance;
405  using std::fill;
406  using std::inner_product;
407  using std::min;
408  using std::minus;
409  using std::size_t;
410 
411  // Initialize f from [data_first, data_last) and fix number of samples
412  f.assign(data_first, data_last);
413  const size_t N = f.size();
414 
415  // Compute the incoming data's mean and centered sum of squares
416  mean = 0;
417  Value sigma2e = 0;
418  welford_nvariance(f.begin(), f.end(), mean, sigma2e);
419 
420  // When requested, subtract the just-computed mean from the data.
421  // Adjust, as necessary, so sigma2e is N times the second moment.
422  if (subtract_mean)
423  {
424  transform(f.begin(), f.end(), f.begin(), bind2nd(minus<Value>(), mean));
425  }
426  else
427  {
428  sigma2e += N*(mean*mean);
429  }
430 
431  // Initialize gain, Dk, and update sigma2e so it is the second moment.
432  Value gain = 1;
433  Value Dk = - f[0]*f[0] - f[N - 1]*f[N - 1] + 2*sigma2e;
434  sigma2e /= N;
435 
436  // At most maxorder N-1 can be fit from N samples. Beware N is unsigned.
437  maxorder = (N == 0) ? 0 : min(static_cast<size_t>(maxorder), N-1);
438 
439  // Output sigma2e and gain for a zeroth order model, if requested.
440  if (hierarchy || maxorder == 0)
441  {
442  *sigma2e_first++ = sigma2e;
443  *gain_first++ = gain;
444  }
445 
446  // Initialize Burg recursion following Collomb
447  if (maxorder) b = f; // Copy iff non-trivial work required
448  Ak.assign(maxorder + 1, Value(0));
449  Ak[0] = 1;
450  ac.clear();
451  ac.reserve(maxorder);
452 
453  // Perform Burg recursion following Collomb
454  for (size_t kp1 = 1; kp1 <= maxorder; ++kp1)
455  {
456  // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
457  // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
458  // By the recurrence, Ak[kp1] will also be the reflection coefficient
459  const Value mu = 2/Dk*inner_product(f.begin() + kp1, f.end(),
460  b.begin(), Value(0));
461  sigma2e *= (1 - mu*mu);
462  for (size_t n = 0; n <= kp1/2; ++n)
463  {
464  Value t1 = Ak[n] - mu*Ak[kp1 - n];
465  Value t2 = Ak[kp1 - n] - mu*Ak[n];
466  Ak[n] = t1;
467  Ak[kp1 - n] = t2;
468  }
469 
470  // Update the gain per Broersen 2006 equation (5.25)
471  gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
472 
473  // Compute and output the next autocorrelation coefficient
474  // See Broersen 2006 equations (5.28) and (5.31) for details
475  ac.push_back(-inner_product(ac.rbegin(), ac.rend(),
476  Ak.begin() + 1, Ak[kp1]));
477 
478  // Output parameters and the input and output variances when requested
479  if (hierarchy || kp1 == maxorder)
480  {
481  params_first = copy(Ak.begin() + 1, Ak.begin() + kp1 + 1,
482  params_first);
483  *sigma2e_first++ = sigma2e;
484  *gain_first++ = gain;
485  }
486 
487  // Update f, b, and then Dk for the next iteration if another remains
488  if (kp1 < maxorder)
489  {
490  for (size_t n = 0; n < N - kp1; ++n)
491  {
492  Value t1 = f[n + kp1] - mu*b[n];
493  Value t2 = b[n] - mu*f[n + kp1];
494  f[n + kp1] = t1;
495  b[n] = t2;
496  }
497  Dk = (1 - mu*mu)*Dk - f[kp1]*f[kp1] - b[N - kp1 - 1]*b[N - kp1 - 1];
498  }
499  }
500 
501  // Output the lag [0,maxorder] autocorrelation coefficients in single pass
502  *autocor_first++ = 1;
503  copy(ac.begin(), ac.end(), autocor_first);
504 
505  // Return the number of values processed in [data_first, data_last)
506  return N;
507 }

1000 template<class RandomAccessIterator,
1001  class InputIterator,
1002  class OutputIterator>
1003 void zohar_linear_solve(RandomAccessIterator a_first,
1004  RandomAccessIterator a_last,
1005  RandomAccessIterator r_first,
1006  InputIterator d_first,
1007  OutputIterator s_first)
1008 {
1009  using std::copy;
1010  using std::distance;
1011  using std::inner_product;
1012  using std::reverse_iterator;
1013 
1014  // Tildes indicate transposes while hats indicate reversed vectors.
1015 
1016  // InputIterator::value_type determines the working precision
1017  typedef typename std::iterator_traits<InputIterator>::value_type value;
1018  typedef typename std::vector<value> vector;
1019  typedef typename vector::size_type size;
1020 
1021  // Determine problem size using [a_first,a_last) and ensure nontrivial
1022  typename std::iterator_traits<RandomAccessIterator>::difference_type dist
1023  = distance(a_first, a_last);
1024  if (dist < 1) throw std::invalid_argument("distance(a_first, a_last) < 1");
1025  const size n = static_cast<size>(dist);
1026 
1027  // Allocate working storage and set initial values for recursion:
1028  vector s; s .reserve(n+1); s .push_back( *d_first);
1029  vector ehat; ehat.reserve(n ); ehat.push_back(-a_first[0]);
1030  vector g; g .reserve(n ); g .push_back(-r_first[0]);
1031  value lambda = 1 - a_first[0]*r_first[0];
1032 
1033  // Though recursive updates to s and g can be done in-place, updates to
1034  // ehat seemingly require one additional vector for storage:
1035  //
1036  // "This sequence of computations is economical of storage. It is only
1037  // necessary to retain quantities computed at level m - 1 until the
1038  // computations at level m are complete." [Trench1967, page 1504]
1039  vector next_ehat; next_ehat.reserve(n);
1040 
1041  // Recursion for i = {1, 2, ..., n - 1}:
1042  for (size i = 1; i < n; ++i)
1043  {
1044 
1045  reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
1046 
1047  // \theta_i = \delta_{i+1} - \tilde{s}_i \hat{r}_i
1048  const value neg_theta = inner_product(
1049  s.begin(), s.end(), rhat_first, value(-(*++d_first)));
1050 
1051  // \eta_i = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
1052  const value neg_eta = inner_product(
1053  ehat.begin(), ehat.end(), a_first, value(a_first[i]));
1054 
1055  // \gamma_i = -\rho_{i+1} - \tilde{g}_i \hat{r}_i
1056  const value neg_gamma = inner_product(
1057  g.begin(), g.end(), rhat_first, value(r_first[i]));
1058 
1059  /*
1060  * s_{i+1} = \bigl(\begin{smallmatrix}
1061  * s_i + (\theta_i/\lambda_i) \hat{e}_i \\
1062  * \theta_i/\lambda_i
1063  * \end{smallmatrix}\bigr)
1064  *
1065  * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
1066  * \eta_i/\lambda_i \\
1067  * \hat{e}_i + (\eta_i/\lambda_i) g_i
1068  * \end{smallmatrix}\bigr)
1069  *
1070  * g_{i+1} = \bigl(\begin{smallmatrix}
1071  * g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
1072  * \gamma_i/\lambda_i
1073  * \end{smallmatrix}\bigr)
1074  */
1075  const value theta_by_lambda = -neg_theta/lambda;
1076  const value eta_by_lambda = -neg_eta /lambda;
1077  const value gamma_by_lambda = -neg_gamma/lambda;
1078  next_ehat.clear();
1079  next_ehat.push_back(eta_by_lambda);
1080  for (size j = 0; j < i; ++j)
1081  {
1082  s[j] += theta_by_lambda*ehat[j];
1083  next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
1084  g[j] += gamma_by_lambda*ehat[j];
1085  }
1086  s.push_back(theta_by_lambda);
1087  g.push_back(gamma_by_lambda);
1088  ehat.swap(next_ehat);
1089 
1090  // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
1091  lambda -= neg_eta*neg_gamma/lambda;
1092  }
1093 
1094  // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
1095  // Computing g_n above was unnecessary but the incremental expense is small
1096  {
1097  reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
1098 
1099  // \theta_n = \delta_{n+1} - \tilde{s}_n \hat{r}_n
1100  const value neg_theta = inner_product(
1101  s.begin(), s.end(), rhat_first, value(-(*++d_first)));
1102 
1103  /*
1104  * s_{n+1} = \bigl(\begin{smallmatrix}
1105  * s_n + (\theta_n/\lambda_n) \hat{e}_n \\
1106  * \theta_n/\lambda_n
1107  * \end{smallmatrix}\bigr)
1108  */
1109  const value theta_by_lambda = -neg_theta/lambda;
1110  for (size j = 0; j < n; ++j)
1111  {
1112  s[j] += theta_by_lambda*ehat[j];
1113  }
1114  s.push_back(theta_by_lambda);
1115  }
1116 
1117  // Output solution
1118  copy(s.begin(), s.end(), s_first);
1119 }
