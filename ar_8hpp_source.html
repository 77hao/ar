00157                         Value&            mean,
00158                         std::size_t&      maxorder,
00159                         OutputIterator1   params_first,
00160                         OutputIterator2   sigma2e_first,
00161                         OutputIterator3   gain_first,
00162                         OutputIterator4   autocor_first,
00163                         const bool        subtract_mean = false,
00164                         const bool        hierarchy = false)

00166     using std::bind2nd;
00167     using std::copy;
00168     using std::distance;
00169     using std::fill;
00170     using std::inner_product;
00171     using std::min;
00172     using std::minus;
00173 
00174     typedef typename std::vector<Value> vector;
00175     typedef typename vector::size_type size;
00176 
00177     // Initialize f from [data_first, data_last) and fix number of samples
00178     vector f(data_first, data_last);
00179     const size N = f.size();
00180 
00181     // Compute the mean of f using pairwise summation and output it.
00182     // Pairwise chosen instead of of Kahan for speed trade off and to avoid
00183     // algorithmic nonsense when working precision is exact (e.g. rationals).
00184     vector b(N, 0);
00185     {
00186         // First pass copies f into b reducing by up to a factor of 2
00187         // Requires b has been adequately sized and filled with zeros
00188         for (size i = 0; i < N; ++i)
00189             b[i/2] += f[i];
00190 
00191         // Initialize i to the next power of 2 lower than N
00192         size t = N, i = !!N;
00193         while (t /= 2)
00194             i *= 2;
00195 
00196         // Recurse on the now power-of-two, smaller problem size
00197         while (i /= 2)
00198             for (size_t j = 0; j < i; ++j)
00199                 b[j] = b[2*j] + b[2*j+1];
00200     }
00201     mean = b[0] / N;
00202 
00203     // Subtract the mean of the data if requested
00204     if (subtract_mean)
00205         transform(f.begin(), f.end(), f.begin(), bind2nd(minus<Value>(), mean));
00206 
00207     // Initialize gain, mean squared discrepancy sigma2e, and Dk.
00208     Value gain = 1;
00209     Value sigma2e = inner_product(f.begin(), f.end(), f.begin(), Value(0));
00210     Value Dk = - f[0]*f[0] - f[N - 1]*f[N - 1] + 2*sigma2e;
00211     sigma2e /= N;
00212 
00213     // Output sigma2e and gain for a zeroth order model, if requested.
00214     if (hierarchy || maxorder == 0)
00215     {
00216         *sigma2e_first++ = sigma2e;
00217         *gain_first++    = gain;
00218     }
00219 
00220     // At most maxorder N-1 can be fit from N samples.  Beware N is unsigned.
00221     maxorder = (N == 0) ? 0 : min(static_cast<size>(maxorder), N-1);
00222 
00223     // Initialize Burg recursion following Collomb
00224     if (maxorder) copy(f.begin(), f.end(), b.begin()); // Copy iff non-trivial
00225     vector Ak(maxorder + 1, Value(0));
00226     Ak[0] = 1;
00227     vector autocor;
00228     autocor.reserve(maxorder);
00229 
00230     // Perform Burg recursion following Collomb
00231     for (size kp1 = 1; kp1 <= maxorder; ++kp1)
00232     {
00233         // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
00234         // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
00235         // By the recurrence, Ak[kp1] will also be the reflection coefficient
00236         const Value mu = 2/Dk*inner_product(f.begin() + kp1, f.end(),
00237                                             b.begin(), Value(0));
00238         sigma2e *= (1 - mu*mu);
00239         for (size n = 0; n <= kp1/2; ++n)
00240         {
00241             Value t1 = Ak[n] - mu*Ak[kp1 - n];
00242             Value t2 = Ak[kp1 - n] - mu*Ak[n];
00243             Ak[n] = t1;
00244             Ak[kp1 - n] = t2;
00245         }
00246 
00247         // Update the gain per Broersen 2006 equation (5.25)
00248         gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
00249 
00250         // Compute and output the next autocorrelation coefficient
00251         // See Broersen 2006 equations (5.28) and (5.31) for details
00252         autocor.push_back(-inner_product(autocor.rbegin(), autocor.rend(),
00253                                          Ak.begin() + 1, Ak[kp1]));
00254 
00255         // Output parameters and the input and output variances when requested
00256         if (hierarchy || kp1 == maxorder)
00257         {
00258             params_first = copy(Ak.begin() + 1, Ak.begin() + kp1 + 1,
00259                                 params_first);
00260             *sigma2e_first++ = sigma2e;
00261             *gain_first++    = gain;
00262         }
00263 
00264         // Update f, b, and then Dk for the next iteration if another remains
00265         if (kp1 < maxorder)
00266         {
00267             for (size n = 0; n < N - kp1; ++n)
00268             {
00269                 Value t1 = f[n + kp1] - mu*b[n];
00270                 Value t2 = b[n] - mu*f[n + kp1];
00271                 f[n + kp1] = t1;
00272                 b[n] = t2;
00273             }
00274             Dk = (1 - mu*mu)*Dk - f[kp1]*f[kp1] - b[N - kp1 - 1]*b[N - kp1 - 1];
00275         }
00276     }
00277 
00278 
00279     // Output the lag [0,maxorder] autocorrelation coefficients in single pass
00280     *autocor_first++ = 1;
00281     copy(autocor.begin(), autocor.end(), autocor_first);
00282 
00283     // Return the number of values processed in [data_first, data_last)
00284     return N;
00285 }

00744 template<class RandomAccessIterator,
00745          class InputIterator,
00746          class OutputIterator>
00747 void zohar_linear_solve(RandomAccessIterator a_first,
00748                         RandomAccessIterator a_last,
00749                         RandomAccessIterator r_first,
00750                         InputIterator        d_first,
00751                         OutputIterator       s_first)
00752 {
00753     using std::copy;
00754     using std::distance;
00755     using std::inner_product;
00756     using std::reverse_iterator;
00757 
00758     // Tildes indicate transposes while hats indicate reversed vectors.
00759 
00760     // InputIterator::value_type determines the working precision
00761     typedef typename std::iterator_traits<InputIterator>::value_type value;
00762     typedef typename std::vector<value> vector;
00763     typedef typename vector::size_type size;
00764 
00765     // Determine problem size using [a_first,a_last) and ensure nontrivial
00766     typename std::iterator_traits<RandomAccessIterator>::difference_type dist
00767             = distance(a_first, a_last);
00768     if (dist < 1) throw std::invalid_argument("distance(a_first, a_last) < 1");
00769     const size n = static_cast<size>(dist);
00770 
00771     // Allocate working storage and set initial values for recursion:
00772     vector s;    s   .reserve(n+1); s   .push_back( *d_first);
00773     vector ehat; ehat.reserve(n  ); ehat.push_back(-a_first[0]);
00774     vector g;    g   .reserve(n  ); g   .push_back(-r_first[0]);
00775     value lambda  = 1 - a_first[0]*r_first[0];
00776 
00777     // Though recursive updates to s and g can be done in-place, updates to
00778     // ehat seemingly require one additional vector for storage:
00779     //
00780     // "This sequence of computations is economical of storage.  It is only
00781     // necessary to retain quantities computed at level m - 1 until the
00782     // computations at level m are complete." [Trench1967, page 1504]
00783     vector next_ehat; next_ehat.reserve(n);
00784 
00785     // Recursion for i = {1, 2, ..., n - 1}:
00786     for (size i = 1; i < n; ++i)
00787     {
00788 
00789         reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
00790 
00791         // \theta_i =  \delta_{i+1}  - \tilde{s}_i \hat{r}_i
00792         const value neg_theta = inner_product(
00793                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
00794 
00795         // \eta_i   = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
00796         const value neg_eta   = inner_product(
00797                 ehat.begin(), ehat.end(), a_first, value(a_first[i]));
00798 
00799         // \gamma_i = -\rho_{i+1}    - \tilde{g}_i \hat{r}_i
00800         const value neg_gamma = inner_product(
00801                 g.begin(), g.end(), rhat_first, value(r_first[i]));
00802 
00803         /*
00804          * s_{i+1} = \bigl(\begin{smallmatrix}
00805          *              s_i + (\theta_i/\lambda_i) \hat{e}_i \\
00806          *              \theta_i/\lambda_i
00807          *          \end{smallmatrix}\bigr)
00808          *
00809          * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
00810          *                     \eta_i/\lambda_i \\
00811          *                     \hat{e}_i + (\eta_i/\lambda_i) g_i
00812          *                 \end{smallmatrix}\bigr)
00813          *
00814          * g_{i+1} = \bigl(\begin{smallmatrix}
00815          *               g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
00816          *               \gamma_i/\lambda_i
00817          *           \end{smallmatrix}\bigr)
00818          */
00819         const value theta_by_lambda = -neg_theta/lambda;
00820         const value   eta_by_lambda = -neg_eta  /lambda;
00821         const value gamma_by_lambda = -neg_gamma/lambda;
00822         next_ehat.clear();
00823         next_ehat.push_back(eta_by_lambda);
00824         for (size j = 0; j < i; ++j)
00825         {
00826             s[j] += theta_by_lambda*ehat[j];
00827             next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
00828             g[j] += gamma_by_lambda*ehat[j];
00829         }
00830         s.push_back(theta_by_lambda);
00831         g.push_back(gamma_by_lambda);
00832         ehat.swap(next_ehat);
00833 
00834         // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
00835         lambda -= neg_eta*neg_gamma/lambda;
00836     }
00837 
00838     // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
00839     // Computing g_n above was unnecessary but the incremental expense is small
00840     {
00841         reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
00842 
00843         // \theta_n =  \delta_{n+1}  - \tilde{s}_n \hat{r}_n
00844         const value neg_theta = inner_product(
00845                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
00846 
00847         /*
00848          * s_{n+1} = \bigl(\begin{smallmatrix}
00849          *              s_n + (\theta_n/\lambda_n) \hat{e}_n \\
00850          *              \theta_n/\lambda_n
00851          *          \end{smallmatrix}\bigr)
00852          */
00853         const value theta_by_lambda = -neg_theta/lambda;
00854         for (size j = 0; j < n; ++j)
00855         {
00856             s[j] += theta_by_lambda*ehat[j];
00857         }
00858         s.push_back(theta_by_lambda);
00859     }
00860 
00861     // Output solution
00862     copy(s.begin(), s.end(), s_first);
00863 }

01740 template <class    Criterion,
01741           typename Integer1,
01742           typename Integer2,
01743           class    InputIterator,
01744           class    OutputIterator>
01745 typename std::iterator_traits<InputIterator>::difference_type
01746 evaluate_models(Integer1       N,
01747                 Integer2       ordfirst,
01748                 InputIterator  first,
01749                 InputIterator  last,
01750                 OutputIterator crit)
01751 {
01752     using std::iterator_traits;
01753 
01754     typedef InputIterator iterator;
01755     typedef typename iterator_traits<iterator>::difference_type difference;
01756     typedef typename iterator_traits<iterator>::value_type      value;
01757 
01758     // Short circuit on trivial input
01759     if (first == last)
01760         return -1;
01761 
01762     // Handle first iteration without comparison as AICC blows up on N == 1
01763     value best_val = evaluate<Criterion>(*first++, N, ordfirst);
01764     difference best_pos = 0, dist = 0;
01765 
01766     // Scan through remainder of candidates updating best as we go
01767     while (first != last)
01768     {
01769 
01770         value candidate = evaluate<Criterion>(*first++, N, ++dist + ordfirst);
01771         *crit++ = candidate;
01772 
01773         if (candidate < best_val) {
01774             best_val = candidate;
01775             best_pos = dist;
01776         }
01777 
01778     }
01779 
01780     return best_pos;
01781 }

01803 template <class    Criterion,
01804           typename Integer,
01805           class    Sequence1,
01806           class    Sequence2,
01807           class    Sequence3,
01808           class    Sequence4,
01809           class    OutputIterator>
01810 typename Sequence1::difference_type
01811 best_model(Integer        N,
01812            Sequence1&     params,
01813            Sequence2&     sigma2e,
01814            Sequence3&     gain,
01815            Sequence4&     autocor,
01816            OutputIterator crit)
01817 {
01818     using std::advance;
01819     using std::copy_backward;
01820 
01821     // Ensure all inputs have conformant sizes
01822     assert(sigma2e.size() > 0);
01823     assert(params .size() == (sigma2e.size()-1)*(sigma2e.size())/2);
01824     assert(gain   .size() == sigma2e.size());
01825     assert(autocor.size() == sigma2e.size());
01826 
01827     // Find the best model index according to the given criterion
01828     const typename Sequence1::difference_type best = evaluate_models<Criterion>(
01829             N, 0u, sigma2e.begin(), sigma2e.end(), crit);
01830 
01831     // Now trim away everything and leaving only the best model in Sequences...
01832 
01833     // ...first in params...
01834     {
01835         // Best parameters might overlap beginning of params so copy_backwards.
01836         // AR(0) is trivial and AR(1) starts at params.begin(), hence off by 1.
01837         typename Sequence1::iterator first  = params.begin();
01838         typename Sequence1::iterator last   = params.begin();
01839         typename Sequence1::iterator result = params.begin();
01840         advance(first,  ((best-1)*best)/2       );
01841         advance(last,   ((best-1)*best)/2 + best);
01842         advance(result,                     best);
01843         copy_backward(first, last, result);
01844         params.resize(best);
01845     }
01846 
01847     // ...next in sigma2e...
01848     {
01849         typename Sequence2::iterator cursor = sigma2e.begin();
01850         advance(cursor, best);
01851         *sigma2e.begin() = *cursor;
01852         sigma2e.resize(1);
01853     }
01854 
01855     // ...next in gain...
01856     {
01857         typename Sequence2::iterator cursor = gain.begin();
01858         advance(cursor, best);
01859         *gain.begin() = *cursor;
01860         gain.resize(1);
01861     }
01862 
01863     // ...and last in autocor...
01864     autocor.resize(best + 1);
01865 
01866     // ...but notice that resizing does not free capacity for std::vectors.
01867 
01868     return best;
01869 }
