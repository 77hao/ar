00144 std::size_t burg_method(InputIterator     data_first,
00145                         InputIterator     data_last,
00146                         Value&            mean,
00147                         std::size_t&      maxorder,
00148                         OutputIterator1   params_first,
00149                         OutputIterator2   sigma2e_first,
00150                         OutputIterator3   gain_first,
00151                         OutputIterator4   autocor_first,
00152                         const bool        subtract_mean = false,
00153                         const bool        hierarchy = false)
00154 {
00155     using std::bind2nd;
00156     using std::copy;
00157     using std::distance;
00158     using std::fill;
00159     using std::inner_product;
00160     using std::min;
00161     using std::minus;
00162 
00163     typedef typename std::vector<Value> vector;
00164     typedef typename vector::size_type size;
00165 
00166     // Initialize f from [data_first, data_last) and fix number of samples
00167     vector f(data_first, data_last);
00168     const size N = f.size();
00169 
00170     // Compute the mean of f using pairwise summation and output it.
00171     // Pairwise chosen instead of of Kahan for speed trade off and to avoid
00172     // algorithmic nonsense when working precision is exact (e.g. rationals).
00173     vector b(N, 0);
00174     {
00175         // First pass copies f into b reducing by up to a factor of 2
00176         // Requires b has been adequately sized and filled with zeros
00177         for (size i = 0; i < N; ++i)
00178             b[i/2] += f[i];
00179 
00180         // Initialize i to the next power of 2 lower than N
00181         size t = N, i = !!N;
00182         while (t /= 2)
00183             i *= 2;
00184 
00185         // Recurse on the now power-of-two, smaller problem size
00186         while (i /= 2)
00187             for (size_t j = 0; j < i; ++j)
00188                 b[j] = b[2*j] + b[2*j+1];
00189     }
00190     mean = b[0] / N;
00191 
00192     // Subtract the mean of the data if requested
00193     if (subtract_mean)
00194         transform(f.begin(), f.end(), f.begin(), bind2nd(minus<Value>(), mean));
00195 
00196     // Initialize gain, mean squared discrepancy sigma2e, and Dk.
00197     Value gain = 1;
00198     Value sigma2e = inner_product(f.begin(), f.end(), f.begin(), Value(0));
00199     Value Dk = - f[0]*f[0] - f[N - 1]*f[N - 1] + 2*sigma2e;
00200     sigma2e /= N;
00201 
00202     // Output sigma2e and gain for a zeroth order model, if requested.
00203     if (hierarchy || maxorder == 0)
00204     {
00205         *sigma2e_first++ = sigma2e;
00206         *gain_first++    = gain;
00207     }
00208 
00209     // At most maxorder N-1 can be fit from N samples.  Beware N is unsigned.
00210     maxorder = (N == 0) ? 0 : min(static_cast<size>(maxorder), N-1);
00211 
00212     // Initialize Burg recursion following Collomb
00213     if (maxorder) copy(f.begin(), f.end(), b.begin()); // Copy iff non-trivial
00214     vector Ak(maxorder + 1, Value(0));
00215     Ak[0] = 1;
00216     vector autocor;
00217     autocor.reserve(maxorder);
00218 
00219     // Perform Burg recursion following Collomb
00220     for (size kp1 = 1; kp1 <= maxorder; ++kp1)
00221     {
00222         // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
00223         // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
00224         // By the recurrence, Ak[kp1] will also be the reflection coefficient
00225         const Value mu = 2/Dk*inner_product(f.begin() + kp1, f.end(),
00226                                             b.begin(), Value(0));
00227         sigma2e *= (1 - mu*mu);
00228         for (size n = 0; n <= kp1/2; ++n)
00229         {
00230             Value t1 = Ak[n] - mu*Ak[kp1 - n];
00231             Value t2 = Ak[kp1 - n] - mu*Ak[n];
00232             Ak[n] = t1;
00233             Ak[kp1 - n] = t2;
00234         }
00235 
00236         // Update the gain per Broersen 2006 equation (5.25)
00237         gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
00238 
00239         // Compute and output the next autocorrelation coefficient
00240         // See Broersen 2006 equations (5.28) and (5.31) for details
00241         autocor.push_back(-inner_product(autocor.rbegin(), autocor.rend(),
00242                                          Ak.begin() + 1, Ak[kp1]));
00243 
00244         // Output parameters and the input and output variances when requested
00245         if (hierarchy || kp1 == maxorder)
00246         {
00247             params_first = copy(Ak.begin() + 1, Ak.begin() + kp1 + 1,
00248                                 params_first);
00249             *sigma2e_first++ = sigma2e;
00250             *gain_first++    = gain;
00251         }
00252 
00253         // Update f, b, and then Dk for the next iteration if another remains
00254         if (kp1 < maxorder)
00255         {
00256             for (size n = 0; n < N - kp1; ++n)
00257             {
00258                 Value t1 = f[n + kp1] - mu*b[n];
00259                 Value t2 = b[n] - mu*f[n + kp1];
00260                 f[n + kp1] = t1;
00261                 b[n] = t2;
00262             }
00263             Dk = (1 - mu*mu)*Dk - f[kp1]*f[kp1] - b[N - kp1 - 1]*b[N - kp1 - 1];
00264         }
00265     }
00266 
00267 
00268     // Output the lag [0,maxorder] autocorrelation coefficients in single pass
00269     *autocor_first++ = 1;
00270     copy(autocor.begin(), autocor.end(), autocor_first);
00271 
00272     // Return the number of values processed in [data_first, data_last)
00273     return N;
00274 }

00318                         RandomAccessIterator a_last,
00319                         RandomAccessIterator r_first,
00320                         InputIterator        d_first,
00321                         OutputIterator       s_first)
00322 {
00323     using std::copy;
00324     using std::distance;
00325     using std::inner_product;
00326     using std::reverse_iterator;
00327 
00328     // Tildes indicate transposes while hats indicate reversed vectors.
00329 
00330     // InputIterator::value_type determines the working precision
00331     typedef typename std::iterator_traits<InputIterator>::value_type value;
00332     typedef typename std::vector<value> vector;
00333     typedef typename vector::size_type size;
00334 
00335     // Determine problem size using [a_first,a_last) and ensure nontrivial
00336     typename std::iterator_traits<RandomAccessIterator>::difference_type dist
00337             = distance(a_first, a_last);
00338     if (dist < 1) throw std::invalid_argument("distance(a_first, a_last) < 1");
00339     const size n = static_cast<size>(dist);
00340 
00341     // Allocate working storage and set initial values for recursion:
00342     vector s;    s   .reserve(n+1); s   .push_back( *d_first);
00343     vector ehat; ehat.reserve(n  ); ehat.push_back(-a_first[0]);
00344     vector g;    g   .reserve(n  ); g   .push_back(-r_first[0]);
00345     value lambda  = 1 - a_first[0]*r_first[0];
00346 
00347     // Though recursive updates to s and g can be done in-place, updates to
00348     // ehat seemingly require one additional vector for storage:
00349     //
00350     // "This sequence of computations is economical of storage.  It is only
00351     // necessary to retain quantities computed at level m - 1 until the
00352     // computations at level m are complete." [Trench1967, page 1504]
00353     vector next_ehat; next_ehat.reserve(n);
00354 
00355     // Recursion for i = {1, 2, ..., n - 1}:
00356     for (size i = 1; i < n; ++i)
00357     {
00358 
00359         reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
00360 
00361         // \theta_i =  \delta_{i+1}  - \tilde{s}_i \hat{r}_i
00362         const value neg_theta = inner_product(
00363                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
00364 
00365         // \eta_i   = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
00366         const value neg_eta   = inner_product(
00367                 ehat.begin(), ehat.end(), a_first, value(a_first[i]));
00368 
00369         // \gamma_i = -\rho_{i+1}    - \tilde{g}_i \hat{r}_i
00370         const value neg_gamma = inner_product(
00371                 g.begin(), g.end(), rhat_first, value(r_first[i]));
00372 
00373         /*
00374          * s_{i+1} = \bigl(\begin{smallmatrix}
00375          *              s_i + (\theta_i/\lambda_i) \hat{e}_i \\
00376          *              \theta_i/\lambda_i
00377          *          \end{smallmatrix}\bigr)
00378          *
00379          * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
00380          *                     \eta_i/\lambda_i \\
00381          *                     \hat{e}_i + (\eta_i/\lambda_i) g_i
00382          *                 \end{smallmatrix}\bigr)
00383          *
00384          * g_{i+1} = \bigl(\begin{smallmatrix}
00385          *               g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
00386          *               \gamma_i/\lambda_i
00387          *           \end{smallmatrix}\bigr)
00388          */
00389         const value theta_by_lambda = -neg_theta/lambda;
00390         const value   eta_by_lambda = -neg_eta  /lambda;
00391         const value gamma_by_lambda = -neg_gamma/lambda;
00392         next_ehat.clear();
00393         next_ehat.push_back(eta_by_lambda);
00394         for (size j = 0; j < i; ++j)
00395         {
00396             s[j] += theta_by_lambda*ehat[j];
00397             next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
00398             g[j] += gamma_by_lambda*ehat[j];
00399         }
00400         s.push_back(theta_by_lambda);
00401         g.push_back(gamma_by_lambda);
00402         ehat.swap(next_ehat);
00403 
00404         // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
00405         lambda -= neg_eta*neg_gamma/lambda;
00406     }
00407 
00408     // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
00409     // Computing g_n above was unnecessary but the incremental expense is small
00410     {
00411         reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
00412 
00413         // \theta_n =  \delta_{n+1}  - \tilde{s}_n \hat{r}_n
00414         const value neg_theta = inner_product(
00415                 s.begin(), s.end(), rhat_first, value(-(*++d_first)));
00416 
00417         /*
00418          * s_{n+1} = \bigl(\begin{smallmatrix}
00419          *              s_n + (\theta_n/\lambda_n) \hat{e}_n \\
00420          *              \theta_n/\lambda_n
00421          *          \end{smallmatrix}\bigr)
00422          */
00423         const value theta_by_lambda = -neg_theta/lambda;
00424         for (size j = 0; j < n; ++j)
00425         {
00426             s[j] += theta_by_lambda*ehat[j];
00427         }
00428         s.push_back(theta_by_lambda);
00429     }
00430 
00431     // Output solution
00432     copy(s.begin(), s.end(), s_first);
00433 }

01297           typename Integer1,
01298           typename Integer2,
01299           class    InputIterator,
01300           class    OutputIterator>
01301 typename std::iterator_traits<InputIterator>::difference_type

01303                 Integer2       ordfirst,
01304                 InputIterator  first,
01305                 InputIterator  last,
01306                 OutputIterator crit)
01307 {
01308     using std::iterator_traits;
01309 
01310     typedef InputIterator iterator;
01311     typedef typename iterator_traits<iterator>::difference_type difference;
01312     typedef typename iterator_traits<iterator>::value_type      value;
01313 
01314     // Short circuit on trivial input
01315     if (first == last)
01316         return -1;
01317 
01318     // Handle first iteration without comparison as AICC blows up on N == 1
01319     value best_val = evaluate<Criterion>(*first++, N, ordfirst);
01320     difference best_pos = 0, dist = 0;
01321 
01322     // Scan through remainder of candidates updating best as we go
01323     while (first != last)
01324     {
01325 
01326         value candidate = evaluate<Criterion>(*first++, N, ++dist + ordfirst);
01327         *crit++ = candidate;
01328 
01329         if (candidate < best_val) {
01330             best_val = candidate;
01331             best_pos = dist;
01332         }
01333 
01334     }
01335 
01336     return best_pos;

01360           typename Integer,
01361           class    Sequence1,
01362           class    Sequence2,
01363           class    Sequence3,
01364           class    Sequence4,
01365           class    OutputIterator>
01366 typename Sequence1::difference_type
01367 best_model(Integer        N,
01368            Sequence1&     params,
01369            Sequence2&     sigma2e,
01370            Sequence3&     gain,
01371            Sequence4&     autocor,
01372            OutputIterator crit)
01373 {
01374     using std::advance;
01375     using std::copy_backward;
01376 
01377     // Ensure all inputs have conformant sizes
01378     assert(sigma2e.size() > 0);
01379     assert(params .size() == (sigma2e.size()-1)*(sigma2e.size())/2);
01380     assert(gain   .size() == sigma2e.size());
01381     assert(autocor.size() == sigma2e.size());
01382 
01383     // Find the best model index according to the given criterion
01384     const typename Sequence1::difference_type best = evaluate_models<Criterion>(
01385             N, 0u, sigma2e.begin(), sigma2e.end(), crit);
01386 
01387     // Now trim away everything and leaving only the best model in Sequences...
01388 
01389     // ...first in params...
01390     {
01391         // Best parameters might overlap beginning of params so copy_backwards.
01392         // AR(0) is trivial and AR(1) starts at params.begin(), hence off by 1.
01393         typename Sequence1::iterator first  = params.begin();
01394         typename Sequence1::iterator last   = params.begin();
01395         typename Sequence1::iterator result = params.begin();
01396         advance(first,  ((best-1)*best)/2       );
01397         advance(last,   ((best-1)*best)/2 + best);
01398         advance(result,                     best);
01399         copy_backward(first, last, result);
01400         params.resize(best);
01401     }
01402 
01403     // ...next in sigma2e...
01404     {
01405         typename Sequence2::iterator cursor = sigma2e.begin();
01406         advance(cursor, best);
01407         *sigma2e.begin() = *cursor;
01408         sigma2e.resize(1);
01409     }
01410 
01411     // ...next in gain...
01412     {
01413         typename Sequence2::iterator cursor = gain.begin();
01414         advance(cursor, best);
01415         *gain.begin() = *cursor;
01416         gain.resize(1);
01417     }
01418 
01419     // ...and last in autocor...
01420     autocor.resize(best + 1);
01421 
01422     // ...but notice that resizing does not free capacity for std::vectors.
01423 
01424     return best;
