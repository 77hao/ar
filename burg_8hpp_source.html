138 template <class InputIterator,
139  class Value,

143  class OutputIterator4>
144 std::size_t burg_method(InputIterator data_first,
145  InputIterator data_last,
146  Value& mean,
147  std::size_t& maxorder,
148  OutputIterator1 params_first,
149  OutputIterator2 sigma2e_first,
150  OutputIterator3 gain_first,
151  OutputIterator4 autocor_first,
152  const bool subtract_mean = false,
153  const bool hierarchy = false)
154 {
155  using std::bind2nd;
156  using std::copy;
157  using std::distance;
158  using std::fill;
159  using std::inner_product;
160  using std::min;
161  using std::minus;
162 
163  typedef typename std::vector<Value> vector;
164  typedef typename vector::size_type size;
165 
166  // Initialize f from [data_first, data_last) and fix number of samples
167  vector f(data_first, data_last);
168  const size N = f.size();
169 
170  // Compute the mean of f using pairwise summation and output it.
171  // Pairwise chosen instead of of Kahan for speed trade off and to avoid
172  // algorithmic nonsense when working precision is exact (e.g. rationals).
173  vector b(N, 0);
174  {
175  // First pass copies f into b reducing by up to a factor of 2
176  // Requires b has been adequately sized and filled with zeros
177  for (size i = 0; i < N; ++i)
178  b[i/2] += f[i];
179 
180  // Initialize i to the next power of 2 lower than N
181  size t = N, i = !!N;
182  while (t /= 2)
183  i *= 2;
184 
185  // Recurse on the now power-of-two, smaller problem size
186  while (i /= 2)
187  for (size_t j = 0; j < i; ++j)
188  b[j] = b[2*j] + b[2*j+1];
189  }
190  mean = b[0] / N;
191 
192  // Subtract the mean of the data if requested
193  if (subtract_mean)
194  transform(f.begin(), f.end(), f.begin(), bind2nd(minus<Value>(), mean));
195 
196  // Initialize gain, mean squared discrepancy sigma2e, and Dk.
197  Value gain = 1;
198  Value sigma2e = inner_product(f.begin(), f.end(), f.begin(), Value(0));
199  Value Dk = - f[0]*f[0] - f[N - 1]*f[N - 1] + 2*sigma2e;
200  sigma2e /= N;
201 
202  // Output sigma2e and gain for a zeroth order model, if requested.
203  if (hierarchy || maxorder == 0)
204  {
205  *sigma2e_first++ = sigma2e;
206  *gain_first++ = gain;
207  }
208 
209  // At most maxorder N-1 can be fit from N samples. Beware N is unsigned.
210  maxorder = (N == 0) ? 0 : min(static_cast<size>(maxorder), N-1);
211 
212  // Initialize Burg recursion following Collomb
213  if (maxorder) copy(f.begin(), f.end(), b.begin()); // Copy iff non-trivial
214  vector Ak(maxorder + 1, Value(0));
215  Ak[0] = 1;
216  vector autocor;
217  autocor.reserve(maxorder);
218 
219  // Perform Burg recursion following Collomb
220  for (size kp1 = 1; kp1 <= maxorder; ++kp1)
221  {
222  // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
223  // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
224  // By the recurrence, Ak[kp1] will also be the reflection coefficient
225  const Value mu = 2/Dk*inner_product(f.begin() + kp1, f.end(),
226  b.begin(), Value(0));
227  sigma2e *= (1 - mu*mu);
228  for (size n = 0; n <= kp1/2; ++n)
229  {
230  Value t1 = Ak[n] - mu*Ak[kp1 - n];
231  Value t2 = Ak[kp1 - n] - mu*Ak[n];
232  Ak[n] = t1;
233  Ak[kp1 - n] = t2;
234  }
235 
236  // Update the gain per Broersen 2006 equation (5.25)
237  gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
238 
239  // Compute and output the next autocorrelation coefficient
240  // See Broersen 2006 equations (5.28) and (5.31) for details
241  autocor.push_back(-inner_product(autocor.rbegin(), autocor.rend(),
242  Ak.begin() + 1, Ak[kp1]));
243 
244  // Output parameters and the input and output variances when requested
245  if (hierarchy || kp1 == maxorder)
246  {
247  params_first = copy(Ak.begin() + 1, Ak.begin() + kp1 + 1,
248  params_first);
249  *sigma2e_first++ = sigma2e;
250  *gain_first++ = gain;
251  }
252 
253  // Update f, b, and then Dk for the next iteration if another remains
254  if (kp1 < maxorder)
255  {
256  for (size n = 0; n < N - kp1; ++n)
257  {
258  Value t1 = f[n + kp1] - mu*b[n];
259  Value t2 = b[n] - mu*f[n + kp1];
260  f[n + kp1] = t1;
261  b[n] = t2;
262  }
263  Dk = (1 - mu*mu)*Dk - f[kp1]*f[kp1] - b[N - kp1 - 1]*b[N - kp1 - 1];
264  }
265  }
266 
267 
268  // Output the lag [0,maxorder] autocorrelation coefficients in single pass
269  *autocor_first++ = 1;
270  copy(autocor.begin(), autocor.end(), autocor_first);
271 
272  // Return the number of values processed in [data_first, data_last)
273  return N;
274 }

315  class InputIterator,
316  class OutputIterator>
317 void zohar_linear_solve(RandomAccessIterator a_first,
318  RandomAccessIterator a_last,
319  RandomAccessIterator r_first,
320  InputIterator d_first,
321  OutputIterator s_first)
322 {
323  using std::copy;
324  using std::distance;
325  using std::inner_product;
326  using std::reverse_iterator;
327 
328  // Tildes indicate transposes while hats indicate reversed vectors.
329 
330  // InputIterator::value_type determines the working precision
331  typedef typename std::iterator_traits<InputIterator>::value_type value;
332  typedef typename std::vector<value> vector;
333  typedef typename vector::size_type size;
334 
335  // Determine problem size using [a_first,a_last) and ensure nontrivial
336  typename std::iterator_traits<RandomAccessIterator>::difference_type dist
337  = distance(a_first, a_last);
338  if (dist < 1) throw std::invalid_argument("distance(a_first, a_last) < 1");
339  const size n = static_cast<size>(dist);
340 
341  // Allocate working storage and set initial values for recursion:
342  vector s; s .reserve(n+1); s .push_back( *d_first);
343  vector ehat; ehat.reserve(n ); ehat.push_back(-a_first[0]);
344  vector g; g .reserve(n ); g .push_back(-r_first[0]);
345  value lambda = 1 - a_first[0]*r_first[0];
346 
347  // Though recursive updates to s and g can be done in-place, updates to
348  // ehat seemingly require one additional vector for storage:
349  //
350  // "This sequence of computations is economical of storage. It is only
351  // necessary to retain quantities computed at level m - 1 until the
352  // computations at level m are complete." [Trench1967, page 1504]
353  vector next_ehat; next_ehat.reserve(n);
354 
355  // Recursion for i = {1, 2, ..., n - 1}:
356  for (size i = 1; i < n; ++i)
357  {
358 
359  reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
360 
361  // \theta_i = \delta_{i+1} - \tilde{s}_i \hat{r}_i
362  const value neg_theta = inner_product(
363  s.begin(), s.end(), rhat_first, value(-(*++d_first)));
364 
365  // \eta_i = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
366  const value neg_eta = inner_product(
367  ehat.begin(), ehat.end(), a_first, value(a_first[i]));
368 
369  // \gamma_i = -\rho_{i+1} - \tilde{g}_i \hat{r}_i
370  const value neg_gamma = inner_product(
371  g.begin(), g.end(), rhat_first, value(r_first[i]));
372 
373  /*
374  * s_{i+1} = \bigl(\begin{smallmatrix}
375  * s_i + (\theta_i/\lambda_i) \hat{e}_i \\
376  * \theta_i/\lambda_i
377  * \end{smallmatrix}\bigr)
378  *
379  * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
380  * \eta_i/\lambda_i \\
381  * \hat{e}_i + (\eta_i/\lambda_i) g_i
382  * \end{smallmatrix}\bigr)
383  *
384  * g_{i+1} = \bigl(\begin{smallmatrix}
385  * g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
386  * \gamma_i/\lambda_i
387  * \end{smallmatrix}\bigr)
388  */
389  const value theta_by_lambda = -neg_theta/lambda;
390  const value eta_by_lambda = -neg_eta /lambda;
391  const value gamma_by_lambda = -neg_gamma/lambda;
392  next_ehat.clear();
393  next_ehat.push_back(eta_by_lambda);
394  for (size j = 0; j < i; ++j)
395  {
396  s[j] += theta_by_lambda*ehat[j];
397  next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
398  g[j] += gamma_by_lambda*ehat[j];
399  }
400  s.push_back(theta_by_lambda);
401  g.push_back(gamma_by_lambda);
402  ehat.swap(next_ehat);
403 
404  // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
405  lambda -= neg_eta*neg_gamma/lambda;
406  }
407 
408  // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
409  // Computing g_n above was unnecessary but the incremental expense is small
410  {
411  reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
412 
413  // \theta_n = \delta_{n+1} - \tilde{s}_n \hat{r}_n
414  const value neg_theta = inner_product(
415  s.begin(), s.end(), rhat_first, value(-(*++d_first)));
416 
417  /*
418  * s_{n+1} = \bigl(\begin{smallmatrix}
419  * s_n + (\theta_n/\lambda_n) \hat{e}_n \\
420  * \theta_n/\lambda_n
421  * \end{smallmatrix}\bigr)
422  */
423  const value theta_by_lambda = -neg_theta/lambda;
424  for (size j = 0; j < n; ++j)
425  {
426  s[j] += theta_by_lambda*ehat[j];
427  }
428  s.push_back(theta_by_lambda);
429  }
430 
431  // Output solution
432  copy(s.begin(), s.end(), s_first);
433 }
