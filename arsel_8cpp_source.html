115  // Look up desired model selection criterion using ar::best_model_function
116  // best_model_function template parameters fit ar::burg_method usage below
117  typedef ar::best_model_function<
118  ar::Burg,size_t,size_t,vector<real>
119  > best_model_function;
120  const best_model_function::type best_model
121  = best_model_function::lookup(criterion, subtract_mean);
122  if (!best_model) {
123  cerr << "Unknown model selection criterion: " << criterion << "\n";
124  return EXIT_FAILURE;
125  }
126 
127  // Use burg_method to estimate a hierarchy of AR models from input data
128  real mu;
129  vector<real> params, sigma2e, gain, autocor;
130  params .reserve(maxorder*(maxorder + 1)/2);
131  sigma2e.reserve(maxorder + 1);
132  gain .reserve(maxorder + 1);
133  autocor.reserve(maxorder + 1);
134  const size_t N = ar::burg_method(istream_iterator<real>(cin),
135  istream_iterator<real>(),
136  mu,
137  maxorder,
138  back_inserter(params),
139  back_inserter(sigma2e),
140  back_inserter(gain),
141  back_inserter(autocor),
142  subtract_mean,
143  /* output hierarchy? */ true);
144 
145 
146  // Keep only best model according to selected criterion
147  best_model(N, minorder, params, sigma2e, gain, autocor);
148 
149  // Compute decorrelation time from the estimated autocorrelation model
150  const real T0 = ar::decorrelation_time(
151  static_cast<size_t>(window_T0*N),
152  ar::autocorrelation(params.begin(), params.end(),
153  gain[0], autocor.begin()),
154  absolute_rho);
155 
156  // Compute effective variance and effective number of independent samples
157  const real eff_var = (N*gain[0]*sigma2e[0]) / (N - T0); // Trenberth1984
158  const real eff_N = N / T0;
159  const real mu_sigma = sqrt(eff_var / eff_N);

164  cout << boolalpha

168  << "\n# eff_var " << eff_var

171  << "\n# minorder " << minorder
172  << "\n# mu " << mu
173  << "\n# mu_sigma " << mu_sigma
174  << "\n# N " << N
175  << "\n# AR(p) " << params.size()
176  << "\n# sigma2eps " << sigma2e[0]
177  << "\n# sigma2x " << gain[0]*sigma2e[0]
178  << "\n# submean " << subtract_mean
179  << "\n# T0 " << T0
180  << "\n# window_T0 " << window_T0

188  return EXIT_SUCCESS;
