33 " Usage: M = arsel (data, submean, absrho, criterion, minorder, maxorder)\n"
34 "\n"
35 " Automatically fit autoregressive models to input signals.\n"
36 "\n"
37 " Use ar::burg_method and ar::best_model to fit an autoregressive process\n"
38 " for signals contained in the rows of matrix data. Sample means will\n"
39 " be subtracted whenever submean is true. Model orders minorder through\n"
40 " min(columns(data), maxorder) will be considered. A dictionary is\n"
41 " returned where each key either contains a result indexable by the\n"
42 " signal number (i.e. the row indices of input matrix data) or it contains\n"
43 " a single scalar applicable to all signals.\n"
44 "\n"
45 " The model order will be selected using the specified criterion.\n"
46 " Criteria are specified using the following abbreviations:\n"
47 " AIC - Akaike information criterion\n"
48 " AICC - asymptotically-corrected Akaike information criterion\n"
49 " BIC - consistent criterion BIC\n"
50 " CIC - combined information criterion\n"
51 " FIC - finite information criterion\n"
52 " FSIC - finite sample information criterion\n"
53 " GIC - generalized information criterion\n"
54 " MCC - minimally consistent criterion\n"
55 "\n"
56 " The number of samples in data (i.e. the number of rows) is returned\n"
57 " in key 'N'. The filter()-ready process parameters are returned\n"
58 " in key 'AR', the sample mean in 'mu', and the innovation variance\n"
59 " \\sigma^2_\\epsilon in 'sigma2eps'. The process output variance\n"
60 " \\sigma^2_\\x and process gain are returned in keys 'sigma2x' and\n"
61 " 'gain', respectively. Autocorrelations for lags zero through the\n"
62 " model order, inclusive, are returned in key 'autocor'. The raw\n"

110  PyErr_SetString(PyExc_RuntimeError,
111  "PyType_Check(ar_ArselType) failed");

116  PyObject *data_obj = NULL;
117  int submean = DEFAULT_SUBMEAN;
118  int absrho = DEFAULT_ABSRHO;
119  const char *criterion = DEFAULT_CRITERION;
120  std::size_t minorder = DEFAULT_MINORDER;
121  std::size_t maxorder = DEFAULT_MAXORDER;
122 
123  // Parse input tuple with second and subsequent arguments optional
124  {
125  unsigned long ul_minorder = minorder, ul_maxorder = maxorder;
126  if (!PyArg_ParseTuple(args, "O|iiskk", &data_obj, &submean, &absrho,
127  &criterion,
128  &ul_minorder, &ul_maxorder)) {
129  return NULL;
130  }
131  minorder = ul_minorder;
132  maxorder = ul_maxorder;
133  }

135  // Lookup the desired model selection criterion
136  typedef ar::best_model_function<
137  ar::Burg,npy_intp,npy_intp,std::vector<double>
138  > best_model_function;
139  const best_model_function::type best_model
140  = best_model_function::lookup(std::string(criterion), submean);
141  if (!best_model) {
142  PyErr_SetString(PyExc_RuntimeError,
143  "Unknown model selection criterion provided to arsel.");
144  return NULL;
145  }
146 
147  // Incoming data may be noncontiguous but should otherwise be well-behaved
148  // On success, 'data' is returned so Py_DECREF is applied only on failure
149  PyObject *data = PyArray_FROMANY(data_obj, NPY_DOUBLE, 1, 2,
150  NPY_ALIGNED | NPY_ELEMENTSTRIDES | NPY_NOTSWAPPED);
151  if (!data) {
152  Py_DECREF(data);
153  return NULL;
154  }
155 
156  // Reshape any 1D ndarray into a 2D ndarray organized as a row vector.
157  // Permits the remainder of the routine to uniformly worry about 2D only.
158  if (1 == ((PyArrayObject *)(data))->nd) {
159  npy_intp dim[2] = { 1, PyArray_DIM(data, 0) };
160  PyArray_Dims newshape = { dim, sizeof(dim)/sizeof(dim[0]) };
161  PyObject *olddata = data;
162  data = PyArray_Newshape((PyArrayObject *)olddata,
163  &newshape, NPY_ANYORDER);
164  Py_DECREF(olddata);
165  if (!data) {
166  Py_DECREF(data);
167  PyErr_SetString(PyExc_RuntimeError,
168  "Unable to reorganize data into matrix-like row vector.");
169  return NULL;
170  }
171  }
172 
173  // How many data points are there?
174  npy_intp M = PyArray_DIM(data, 0);
175  npy_intp N = PyArray_DIM(data, 1);
176 
177  // Prepare per-signal storage locations to return to caller
178  // TODO Ensure these invocations all worked as expected
179  PyObject *_AR = PyList_New(M);
180  PyObject *_autocor = PyList_New(M);
181  PyObject *_eff_N = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
182  PyObject *_eff_var = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
183  PyObject *_gain = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
184  PyObject *_mu = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
185  PyObject *_mu_sigma = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
186  PyObject *_sigma2eps = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
187  PyObject *_sigma2x = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
188  PyObject *_T0 = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
189 
190  // Prepare vectors to capture burg_method() output
191  std::vector<double> params, sigma2e, gain, autocor;
192  params .reserve(maxorder*(maxorder + 1)/2);
193  sigma2e.reserve(maxorder + 1);
194  gain .reserve(maxorder + 1);
195  autocor.reserve(maxorder + 1);
196 
197  // Prepare repeatedly-used working storage for burg_method()
198  std::vector<double> f, b, Ak, ac;
199 
200  // Process each signal in turn...
201  for (npy_intp i = 0; i < M; ++i)
202  {
203  // Use burg_method to estimate a hierarchy of AR models from input data
204  params .clear();
205  sigma2e.clear();
206  gain .clear();
207  autocor.clear();
208  ar::strided_adaptor<const double*> signal_begin(
209  (const double*) PyArray_GETPTR2(data, i, 0),
210  PyArray_STRIDES(data)[1] / sizeof(double));
211  ar::strided_adaptor<const double*> signal_end (
212  (const double*) PyArray_GETPTR2(data, i, N),
213  PyArray_STRIDES(data)[1] / sizeof(double));
214  ar::burg_method(signal_begin, signal_end,
215  *(double*)PyArray_GETPTR1(_mu, i),
216  maxorder,
217  std::back_inserter(params),
218  std::back_inserter(sigma2e),
219  std::back_inserter(gain),
220  std::back_inserter(autocor),
221  submean, /* output hierarchy? */ true, f, b, Ak, ac);
222 
223  // Keep only best model per chosen criterion via function pointer
224  try {
225  best_model(N, minorder, params, sigma2e, gain, autocor);
226  }
227  catch (std::exception &e)
228  {
229  PyErr_SetString(PyExc_RuntimeError, e.what());
230  goto fail;
231  }
232 
233  // Compute decorrelation time from the estimated autocorrelation model
234  ar::predictor<double> p = ar::autocorrelation(
235  params.begin(), params.end(), gain[0], autocor.begin());
236  *(double*)PyArray_GETPTR1(_T0, i)
237  = ar::decorrelation_time(N, p, absrho);
238 
239  // Filter()-ready process parameters in field 'AR' with leading one
240  PyObject *_ARi = PyList_New(params.size() + 1);
241  PyList_SET_ITEM(_AR, i, _ARi);
242  PyList_SET_ITEM(_ARi, 0, PyFloat_FromDouble(1));
243  for (std::size_t k = 0; k < params.size(); ++k) {
244  PyList_SET_ITEM(_ARi, k+1, PyFloat_FromDouble(params[k]));
245  }
246 
247  // Field 'sigma2eps'
248  *(double*)PyArray_GETPTR1(_sigma2eps, i) = sigma2e[0];
249 
250  // Field 'gain'
251  *(double*)PyArray_GETPTR1(_gain, i) = gain[0];
252 
253  // Field 'sigma2x'
254  *(double*)PyArray_GETPTR1(_sigma2x, i) = gain[0]*sigma2e[0];
255 
256  // Field 'autocor'
257  PyObject *_autocori = PyList_New(autocor.size());
258  PyList_SET_ITEM(_autocor, i, _autocori);
259  for (std::size_t k = 0; k < autocor.size(); ++k) {
260  PyList_SET_ITEM(_autocori, k, PyFloat_FromDouble(autocor[k]));
261  }
262 
263  // Field 'eff_var'
264  // Unbiased effective variance expression from [Trenberth1984]
265  *(double*)PyArray_GETPTR1(_eff_var, i)
266  = (N*gain[0]*sigma2e[0]) / (N - *(double*)PyArray_GETPTR1(_T0, i));
267 
268  // Field 'eff_N'
269  *(double*)PyArray_GETPTR1(_eff_N, i)
270  = N / *(double*)PyArray_GETPTR1(_T0, i);
271 
272  // Field 'mu_sigma'
273  // Variance of the sample mean using effective quantities
274  *(double*)PyArray_GETPTR1(_mu_sigma, i)
275  = std::sqrt( *(double*)PyArray_GETPTR1(_eff_var, i)
276  / *(double*)PyArray_GETPTR1(_eff_N, i));
277  }
